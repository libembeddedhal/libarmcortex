261 template<size_t VectorCount>
262 static void initialize()
263 {
264 // Statically allocate a buffer of vectors to be used as the new IVT.
265 static constexpr size_t total_vector_count = VectorCount + core_interrupts;
266 alignas(512) static std::array<interrupt_pointer, total_vector_count>
267 vector_buffer{};
268
269 if (system_control().get_interrupt_vector_table_address() == nullptr) {
270 // Assign the statically allocated vector within this scope to the global
271 // vector_table span so that it can be accessed in other
272 // functions. This is valid because the interrupt vector table has static
273 // storage duration and will exist throughout the duration of the
274 // application.
275 vector_table = vector_buffer;
276
277 // Will fill the interrupt handler and vector table with a function that
278 // does nothing.
279 std::fill(vector_buffer.begin(), vector_buffer.end(), nop);
280
281 // Relocate the interrupt vector table the vector buffer. By default this
282 // will be set to the address of the start of flash memory for the MCU.
283 system_control().set_interrupt_vector_table_address(vector_buffer.data());
284 }
285 }

300 // interrupt vectors.

382 [[nodiscard]] boost::leaf::result<bool> verify_vector_enabled(
383 interrupt_pointer p_handler)
384 {
385 BOOST_LEAF_CHECK(sanity_check());
386
387 // Check if the handler match
388 auto irq_handler = vector_table[m_irq.vector_index()];
389 bool handlers_are_the_same = (irq_handler == p_handler);
390
391 if (!handlers_are_the_same) {
392 return false;
393 }
394
395 if (m_irq.default_enabled()) {
396 return true;
397 }
398
399 uint32_t enable_register = nvic()->iser.at(m_irq.register_index());
400 return (enable_register & m_irq.enable_mask()) == 0U;
401 }

408 }

412 }
