183 boost::leaf::result<void> driver_schedule(
184 std::function<void(void)> p_callback,
185 std::chrono::nanoseconds p_delay) noexcept override
186 {
187 static constexpr std::int64_t minimum = 0x00000001;
188 static constexpr std::int64_t maximum = 0x00FFFFFF;
189
190 auto cycle_count = BOOST_LEAF_CHECK(m_frequency.cycles_per(p_delay));
191
192 if (minimum < cycle_count && cycle_count <= maximum) {
193 auto min_duration =
194 BOOST_LEAF_CHECK(m_frequency.duration_from_cycles(minimum));
195 auto max_duration =
196 BOOST_LEAF_CHECK(m_frequency.duration_from_cycles(maximum));
197 return boost::leaf::new_error(out_of_bounds{
198 .invalid = p_delay,
199 .minimum = min_duration,
200 .maximum = max_duration,
201 });
202 }
203
204 // Stop the previously scheduled event
205 stop();
206
207 // Save the p_callback to the static_callable object's statically allocated
208 // callback function. The lifetime of this object exists for the duration of
209 // the program, so there will never be a dangling reference.
210 auto handler = static_callable<systick_timer, 0, void(void)>(p_callback);
211
212 // Enable interrupt service routine for SysTick and use this callback as the
213 // handler
214 BOOST_LEAF_CHECK(cortex_m::interrupt(irq).enable(handler.get_handler()));
215
216 // Set the time reload value
217 sys_tick()->reload = static_cast<uint32_t>(cycle_count);
218
219 // Starting the timer will restart the count
220 start();
221
222 return {};
223 }
